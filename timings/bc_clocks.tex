\def\Dimline[#1][#2][#3][#4]{
    \begin{scope}[>=latex] % redef arrow for dimension lines
        \draw let \p1=#1, \p2=#2, \n0={veclen(\x2-\x1,\y2-\y1)} in [|<->|,
        decoration={markings, % switch on markings
                mark=at position .5 with {\node[#3] at (0,0) {#4};},
        },
        postaction=decorate] #1 -- #2 ;
    \end{scope}
}
